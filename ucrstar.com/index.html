// Apply selected color option to map layer (Option 2)
const colorOption = {
  fill: ['#7ec8c0', '#1976d2', '#8b9f6e'],
  outline: '#2d5a4a'
};

map.setPaintProperty('buildings-fill', 'fill-color', colorOption.fill[0]);
map.setPaintProperty('buildings-fill', 'fill-outline-color', colorOption.outline);

// Optional: cycle fill by attribute for multi-color
map.setPaintProperty('buildings-fill', 'fill-color', [
  'match', ['get', 'type'],
  'residential', colorOption.fill[0],
  'commercial', colorOption.fill[1],
  colorOption.fill[2]
]);
